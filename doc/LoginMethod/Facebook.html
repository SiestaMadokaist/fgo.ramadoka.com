def user_data
  # prefer to use ._user_data for local access
  # use .user_data for global access
  # not doing so might lead to mutual recursion
  ud = _user_data
  Hash[
    facebook: ud[:id],
    avatar_url: avatar_url,
    name: ud[:name],
    facebook_id: ud[:id]
  ]
end
