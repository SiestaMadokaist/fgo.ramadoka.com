def validate_password_change!(options = {})
  @password_hashed = true
  pc = Component::User::PasswordChanger.new(
    user: self,
    uid: options[:uid]
  )
  pc.validate!(validation: options[:validation])
  self.password = pc.password.value
  self.save!
end
