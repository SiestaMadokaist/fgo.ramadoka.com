def readable(format = _default_readable_format, formatter = _default_readable_formatter)
  sfmt = format.gsub(/%{(\w+)}/){|match| "#\{formatter[:#{$1}]\}" }
  mmt = eval("\"#{sfmt}\"")
  keys = []
  format.scan(/%\{(\w+)\}/){|match| keys << eval(match[0]) }
  mmt % keys
end
